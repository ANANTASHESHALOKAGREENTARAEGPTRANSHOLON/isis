  boolean hasLimit()

  long getEnd()     // <.>
  int getStartAsInt()     // <.>
  int getLimitAsInt()     // <.>

  QueryRange unconstrained()

  QueryRange limit(long limit)
  QueryRange of(long... range)
  QueryRange withStart(long start)
  QueryRange withLimit(long limit)
}
